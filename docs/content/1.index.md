```ts [Composition API]
useSchemaOrg([
  definePerson({
    name: 'Harlan Wilton',
    image: '/me.png',
    sameAs: [
      'https://github.com/harlan-zw',
      'https://twitter.com/harlan_zw'
    ]
  }),
  defineWebSite({ name: 'My Blog' }),
  defineWebPage(),
])
```

```vue [Component API]
<template>
  <SchemaOrgPerson
    name="Harlan Wilton"
    image="/me.png"
    :same-as="[
      'https://github.com/harlan-zw',
      'https://twitter.com/harlan_zw'
    ]"
  />
  <SchemaOrgWebSite name="My Blog" />
  <SchemaOrgWebPage />
</template>
```
